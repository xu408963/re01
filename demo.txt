    "clear":"清楚"
    "pwd":"显示当前终端位置"，
    "ls":"显示当前目录下的文件"，
    "cd":"切换路径，后面跟路径，..是返回上一级"
    "git init":"初始化"
    "git add+文件夹名字"："加入暂存区"
    "git add.":"当前目录下的全部文件夹"
    "git commit":"提交把add保存的固定成一个版本"
    "wq":"保存退出"
    "git log":"察看"
    "git commit -m'提交名字'"："简化版操作"
    "git reset --hard +id":"返回指定的id操作（!不可逆）"
    "git branch":"创建分支"
    "git checkout":"切换分支"
